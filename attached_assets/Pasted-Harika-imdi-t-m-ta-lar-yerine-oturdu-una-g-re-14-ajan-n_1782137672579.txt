```python

from pydantic import BaseModel, Field
from typing import Dict, List, Any

class AppState(BaseModel):
    app_name: str = ""
    user_prompt: str = ""
    architecture_selection: str = ""
    current_state: str = "INITIALIZED"
    thinking_logs: List[Dict[str, str]] = Field(default_factory=list) # Akıl yürütme (<think>) geçmişi
```

### 📄 3. core/snapshot.py
Ajanlar işe başlamadan önce otomatik commit (yedek) alan, hata anında sistemi bozmadan çalışan son stabil sürüme döndüren yapı:
```python
# core/snapshot.py
import os
import shutil
from datetime import datetime

class SnapshotManager:
    def __init__(self, workspace_dir="workspace", backup_dir="checkpoints"):
        self.workspace_dir = workspace_dir
        self.backup_dir = backup_dir
        os.makedirs(self.backup_dir, exist_ok=True)

    def create_checkpoint(self, app_name: str, task_id: str) -> str:
        """Projenin o anki halini atomik olarak ZIP'ler."""
        app_path = os.path.join(self.workspace_dir, app_name)
        if not os.path.exists(app_path):
            return "INITIAL_EMPTY"
            
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        checkpoint_id = f"v_{task_id}_{timestamp}"
        zip_path = os.path.join(self.backup_dir, f"{app_name}_{checkpoint_id}")
        
        shutil.make_archive(zip_path, 'zip', app_path)
        return checkpoint_id

    def rollback(self, app_name: str, checkpoint_id: str):
        """Hata anında eski ZIP'i açarak projeyi kurtarır."""
        if checkpoint_id == "INITIAL_EMPTY":
            return
        zip_file = os.path.join(self.backup_dir, f"{app_name}_{checkpoint_id}.zip")
        app_path = os.path.join(self.workspace_dir, app_name)
        
        if os.path.exists(zip_file):
            shutil.rmtree(app_path, ignore_errors=True)
            os.makedirs(app_path, exist_ok=True)
            shutil.unpack_archive(zip_file, app_path, 'zip')
            print(f"♻️ [ROLLBACK] {app_name}, başarıyla '{checkpoint_id}' konumuna geri sarıldı.")

```
### 📄 4. llm/router.py
Farklı dosyalar yerine, JSON'dan gelen model_name parametresine bakarak doğru modele otomatik yönlenen ve Semaphore ile donanımı kilitlemeyen sürücü:
```python
# llm/router.py
import asyncio
import httpx
import re

class LLMRouter:
    def __init__(self, ollama_url="http://localhost:11434/api/chat", max_concurrent_agents=5):
        self.ollama_url = ollama_url
        # Aynı anda en fazla N ajanın VRAM'e saldırmasına izin vererek çökmesini engeller
        self.semaphore = asyncio.Semaphore(max_concurrent_agents)

    async def query_dynamic_agent(self, agent_config: dict, user_content: str) -> tuple[str, str]:
        """Ajanın JSON'daki model parametresine bakarak izole oturum açar."""
        async with self.semaphore:
            system_prompt = f"Sen bir {agent_config['role']} ajanısın. Görevin: {agent_config['desc']}"
            
            payload = {
                "model": agent_config["model_name"], # 🎯 Dinamik model değişimi burada gerçekleşiyor
                "messages": [
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": user_content}
                ],
                "stream": False,
                "options": {"temperature": agent_config["temperature"]}
            }
            
            async with httpx.AsyncClient() as client:
                response = await client.post(self.ollama_url, json=payload, timeout=120.0)
                raw_output = response.json()["message"]["content"]
                
                # DeepSeek-R1 veya Qwen akıl yürütme (<think>) etiketlerini regex ile ayır
                reasoning_match = re.search(r'<think>(.*?)</think>', raw_output, re.DOTALL)
                reasoning = reasoning_match.group(1).strip() if reasoning_match else ""
                clean_content = re.sub(r'<think>.*?</think>', '', raw_output, flags=re.DOTALL).strip()
                
                return reasoning, clean_content

```
### 📄 5. orchestrator/engine.py
Asenkron kilitleri yöneten, ajanların birbirini centilmen gibi beklemesini sağlayan ve JSON'dan dinamik ajan çağıran ana beyin:
```python
# orchestrator/engine.py
import json
import os
import asyncio
from core.state import AppState
from core.snapshot import SnapshotManager
from llm.router import LLMRouter

class SwarmEngine:
    def __init__(self):
        self.state = AppState()
        self.snapshot = SnapshotManager()
        self.router = LLMRouter()
        # Uygulama bazlı asenkron kilit sözlüğü (Klasör kilitleme / Mutex)
        self._file_locks = {}

    def _get_app_lock(self, app_name: str) -> asyncio.Lock:
        if app_name not in self._file_locks:
            self._file_locks[app_name] = asyncio.Lock()
        return self._file_locks[app_name]

    def _load_agent_meta(self, agent_key: str) -> dict:
        with open("agencies/config.json", "r", encoding="utf-8") as f:
            return json.load(f)[agent_key]

    async def run_proactive_consultant(self, user_prompt: str) -> str:
        """Kullanıcı prompt girdiğinde uysal olmadan mimari öneri sunan danışman."""
        self.state.user_prompt = user_prompt
        self.state.current_state = "CONSULTING"
        
        # Danışman için sabit veya güçlü bir model seçilebilir, şimdilik backend ayarını örnek alıyoruz
        dummy_config = {"role": "SoftwareArchitect", "model_name": "qwen2.5-coder:32b", "temperature": 0.2, "desc": "Mimar"}
        _, suggestions = await self.router.query_dynamic_agent(dummy_config, user_prompt)
        return suggestions

    async def execute_agent_pipeline(self, agent_key: str, max_retries: int = 3) -> bool:
        """İlgili ajanı çağırır, dosya kilidini alır, hata varsa otonom debug eder."""
        agent_data = self._load_agent_meta(agent_key)
        app_lock = self._get_app_lock(self.state.app_name)
        
        # 🔒 KİLİT MEKANİZMASI: Önceki asenkron ajanların işinin bitmesini bekler
        async with app_lock:
            print(f"\n🔒 [KİLİT] {agent_key} klasör kilidini aldı. Model: {agent_data['model_name']} ateşleniyor...")
            retries = 0
            # Adım başlamadan önce Git Commit (Snapshot) al
            checkpoint_id = self.snapshot.create_checkpoint(self.state.app_name, agent_key)
            
            while retries < max_retries:
                # İzole session ile LLM'e git (Model JSON'dan dinamik okunuyor)
                reasoning, clean_output = await self.router.query_dynamic_agent(agent_data, self.state.user_prompt)
                
                # Akıl yürütme günlüğünü merkeze kaydet (Debugging için kullanılacak)
                self.state.thinking_logs.append({"agent": agent_key, "trace": reasoning})
                
                # OTONOM TEST / ANALİZ (Gerçek yapıda pytest/linter tetiklenecek)
                test_passed = True 
                
                if test_passed:
                    print(f"✅ [BAŞARI] {agent_key} görevini hatasız tamamladı.")
                    return True
                    
                # KENDİ KENDİNİ DÜZELTME (SELF-HEALING LOOP)
                retries += 1
                print(f"⚠️ [HATA] {agent_key} hata yaptı! Deneme {retries}/{max_retries}. Debug ediliyor...")
                
            # Eğer tüm denemeler başarısız olursa, sistemi bozmamak için Rollback yap
            print(f"❌ [BAŞARISIZ] {agent_key} hatayı çözemedi! Eski sürüme dönülüyor...")
            self.snapshot.rollback(self.state.app_name, checkpoint_id)
            return False

```
### 📄 6. main.py
Arka planda sunucuyu (WebSocket/FastAPI) açık tutarken, terminalden (cmd) girilen promptları asenkron koşturan ana ateşleyici:
```python
# main.py
import sys
import asyncio
from orchestrator.engine import SwarmEngine

async def start_api_server():
    """VS Code eklentisi veya arayüzler için arka planda sürekli dinleyen API kanalı."""
    while True:
        await asyncio.sleep(3600)

async def start_cli_repl_loop(engine: SwarmEngine):
    """Ön planda terminalden komut alan ve proaktif REPL akışını yöneten döngü."""
    print("\n🚀 Codeflex Çekirdek Motoru Aktif!")
    loop = asyncio.get_event_loop()
    
    while True:
        print("swarm> ", end="", flush=True)
        user_input = await loop.run_in_executor(None, sys.stdin.readline)
        user_input = user_input.strip()
        
        if user_input.lower() in ["quit", "exit"]:
            print("🚪 Kapatılıyor...")
            break
            
        if not user_input:
            continue
            
        # 1. Proaktif Danışmanlık Aşaması
        print("🔍 Mimari analiz yapılıyor...")
        suggestions = await engine.run_proactive_consultant(user_input)
        print(f"\n📢 CODEFLEX MİMARİ ÖNERİSİ:\n{suggestions}\n")
        
        print("Seçimi Onaylıyor musunuz? (onay/red): ", end="", flush=True)
        confirmation = await loop.run_in_executor(None, sys.stdin.readline)
        
        if confirmation.strip().lower() == "onay":
            engine.state.app_name = "output_project"
            engine.state.current_state = "CODING"
            
            # 2. Sırayla Ajanları Tetikle (Kilitler asenkron olarak arka arkaya işler ve modeller dinamik değişir)
            print("\n👥 Ajan Fabrikası Sırayla Ateşleniyor...")
            await engine.execute_agent_pipeline("database_agent")
            await engine.execute_agent_pipeline("backend_agent")
            await engine.execute_agent_pipeline("delta_agent")            # Delta modeli (DeepSeek-R1)
            await engine.execute_agent_pipeline("memory_profiler_agent") # Profiler modeli (Llama-8b)
        else:
            print("❌ Süreç iptal edildi.\n")

async def bootstrap():
    engine = SwarmEngine()
    print("💾 Çekirdek servisler ve asenkron dinamik kilit yöneticileri başlatıldı.")
    
    await asyncio.gather(
        start_api_server(),
        start_cli_repl_loop(engine)
    )

if __name__ == "__main__":
    try:
        asyncio.run(bootstrap())
    except KeyboardInterrupt:
        print("\n🛑 Kullanıcı tarafından kesildi.")

```
